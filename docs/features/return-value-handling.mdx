Handler return values are captured in `EventResult` records and can be consumed as a single value or aggregated across handlers.

Repository example files:
- [`examples/simple.py`](https://github.com/ArchiveBox/abxbus/blob/main/examples/simple.py)
- [`abxbus-ts/examples/simple.ts`](https://github.com/ArchiveBox/abxbus/blob/main/abxbus-ts/examples/simple.ts)
- [`abxbus-rust/tests/test_typed_events.rs`](https://github.com/ArchiveBox/abxbus/blob/main/abxbus-rust/tests/test_typed_events.rs)

## Typed return values

Use the event result type to enforce return typing across handlers.

<Tabs>
<Tab title="Python">

```python
from abxbus import BaseEvent, EventBus

class DoMathEvent(BaseEvent[int]):
    a: int
    b: int

def add(event: DoMathEvent) -> int:
    return event.a + event.b

bus = EventBus('AppBus')
bus.on(DoMathEvent, add)

event = await bus.emit(DoMathEvent(a=2, b=3))
result = await event.event_result()
```

</Tab>
<Tab title="TypeScript">

```ts
import { BaseEvent, EventBus } from 'abxbus'
import { z } from 'zod'

const DoMathEvent = BaseEvent.extend('DoMathEvent', {
  a: z.number(),
  b: z.number(),
  event_result_type: z.number(),
})

const bus = new EventBus('AppBus')
bus.on(DoMathEvent, (event) => event.a + event.b)

const event = bus.emit(DoMathEvent({ a: 2, b: 3 }))
await event.done()
const result = event.event_result
```

</Tab>
<Tab title="Rust">

```rust
use abxbus_rust::{
    event,
    event_bus::EventBus,
    BaseEvent,
};
use futures::executor::block_on;

event! {
    struct DoMathEvent {
        a: i64,
        b: i64,
        event_result_type: i64,
    }
}

let bus = EventBus::new(Some("AppBus".to_string()));
bus.on::<DoMathEvent, _, _>("add", |event| async move {
    let payload = event.event_payload();
    Ok(payload.a + payload.b)
});

let event = bus.emit(DoMathEvent { a: 2, b: 3 });
let result = block_on(event.event_result(Default::default()))?;
```

</Tab>
<Tab title="Go">

```go
type DoMathPayload struct {
	A int `json:"a"`
	B int `json:"b"`
}

bus := abxbus.NewEventBus("AppBus", nil)
abxbus.OnTyped[DoMathPayload, int](
	bus,
	"DoMathEvent",
	"add",
	func(ctx context.Context, payload DoMathPayload) (int, error) {
		return payload.A + payload.B, nil
	},
	nil,
)

event, err := abxbus.NewBaseEventWithResult[DoMathPayload, int](
	"DoMathEvent",
	DoMathPayload{A: 2, B: 3},
)
if err != nil {
	panic(err)
}
result, err := bus.Emit(event).EventResult(context.Background())
```

</Tab>
</Tabs>

## Aggregating multiple handler results

When multiple handlers respond to the same event, collect all successful values with the list helpers.

<Tabs>
<Tab title="Python">

```python
event = await bus.emit(GetConfigEvent())
values = await event.event_results_list(raise_if_any=False, raise_if_none=False)
```

</Tab>
<Tab title="TypeScript">

```ts
const event = bus.emit(GetConfigEvent({}))
const values = await event.eventResultsList({ raise_if_any: false, raise_if_none: false })
```

</Tab>
<Tab title="Rust">

```rust
use abxbus_rust::{
    base_event::EventResultsOptions,
    event,
    event_bus::EventBus,
    BaseEvent,
};
use futures::executor::block_on;
use serde_json::json;

event! {
    struct GetConfigEvent {
        event_result_type: serde_json::Value,
    }
}

let bus = EventBus::new(Some("ConfigBus".to_string()));

bus.on::<GetConfigEvent, _, _>("user_config", |_event| async move {
    Ok(json!({"debug": true, "port": 8080}))
});
bus.on::<GetConfigEvent, _, _>("system_config", |_event| async move {
    Ok(json!({"debug": false, "timeout": 30}))
});

let event = bus.emit(GetConfigEvent {});
let values = block_on(event.event_results_list(EventResultsOptions {
    raise_if_any: false,
    raise_if_none: false,
    timeout: None,
}))?;
```

</Tab>
<Tab title="Go">

```go
event := bus.Emit(abxbus.NewBaseEvent("GetConfigEvent", nil))
values, err := event.EventResultsList(
	context.Background(),
	nil,
	&abxbus.EventResultsListOptions{RaiseIfAny: false, RaiseIfNone: false},
)
```

</Tab>
</Tabs>

## Result list options

<Tabs>
<Tab title="Python">

```python
await event.event_results_list(raise_if_any=False, raise_if_none=False)
await event.event_results_list(include=lambda event_result: isinstance(event_result.result, dict), raise_if_any=False)
await event.event_results_list(timeout=0.25)
```

</Tab>
<Tab title="TypeScript">

```ts
await event.eventResultsList({ raise_if_any: false, raise_if_none: false })
await event.eventResultsList((result) => typeof result === 'object', { raise_if_any: false })
await event.eventResultsList({ timeout: 0.25 })
```

</Tab>
<Tab title="Rust">

```rust
use abxbus_rust::base_event::EventResultsOptions;

event.event_results_list(EventResultsOptions {
    raise_if_any: false,
    raise_if_none: false,
    timeout: None,
}).await?;
event.event_results_list(EventResultsOptions {
    timeout: Some(0.25),
    ..EventResultsOptions::default()
}).await?;
```

</Tab>
<Tab title="Go">

```go
event.EventResultsList(ctx, nil, &abxbus.EventResultsListOptions{
	RaiseIfAny: false,
	RaiseIfNone: false,
})
timeout := 0.25
event.EventResultsList(ctx, nil, &abxbus.EventResultsListOptions{Timeout: &timeout})
```

</Tab>
</Tabs>

- `raise_if_any`: raise if any handler ended with an error.
- `raise_if_none`: raise only when no handlers returned a valid value after filtering; it does not raise just because one handler returned `None`/`null`/`undefined`.
- Default filtering includes only completed, successful, non-empty scalar/object/list values and excludes forwarded `BaseEvent` returns.

## Per-handler inspection

All runtimes keep per-handler result metadata in addition to the aggregate result helpers.

<Tabs>
<Tab title="Python">

```python
by_name = {result.handler_name: result.result for result in event.event_results.values()}
```

</Tab>
<Tab title="TypeScript">

```ts
const byHandler = event.event_results
```

</Tab>
<Tab title="Rust">

```rust
let by_name: std::collections::HashMap<_, _> = event
    .inner
    .lock()
    .event_results
    .values()
    .map(|result| (result.handler.handler_name.clone(), result.result.clone()))
    .collect();
```

</Tab>
<Tab title="Go">

```go
byName := map[string]any{}
for _, result := range event.EventResults {
	byName[result.HandlerName] = result.Result
}
```

</Tab>
</Tabs>
